Actions

Une action = une méthode définie dans le prototype du générateur


yeoman.generators.base.extend({
	step1 : function () {},
	step2 : function () {},
	step3 : function () {}
});

Chaque action est appelée dans l'ordre de déclaration

Définir une méthode qui ne sera pas appelée


yeoman.generators.base.extend({
	step1 : function () {},
	step2 : function () {},
	step3 : function () {
		this._stepNotCalled();
	},
	_stepNotCalled : function () {}
});

Prompts

Inquirer.js 

yeoman.generators.Base.extend({
	prompt: function () {
		var done = this.async();
		this.prompt([{
			type    : "input",
			name    : "name",
			message : "Your project name",
			default : "A Cool project"
		}], function (answers) {
			console.log(answers.name);
			done();
		});
	}
})
