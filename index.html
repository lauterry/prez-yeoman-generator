Actions

Une action est automatiquement appelée par le générateur ...

... si elle est définie dans le constructeur de celui-ci


yeoman.generators.base.extend({

	step1 : function () {
		// action 1
	},

	step2 : function () {
		// action 2
	},

	step3 : function () {
		// action 3
	}

});

Chaque action est appelée dans l'ordre de déclaration

Définir une méthode qui ne sera pas appelée ?


yeoman.generators.base.extend({

	step1 : function () {},

	step2 : function () {},

	step3 : function () {
		this._stepNotCalled();
	},

	_stepNotCalled : function () {}

});

yeoman.generators.Base.extend({

	prompt: function () {

		var done = this.async();

		this.prompt([{
			type    : "input",
			name    : "name",
			message : "Your project name",
			default : "A Cool project"
		}], function (answers) {
			console.log(answers.name);
			done();
		});

	}

})
